//@version=5
indicator("Multi MA System", shorttitle="MultiMA", overlay=true)

// Input Settings
ma1_length = input.int(9, "Fast EMA", minval=1, maxval=200)
ma2_length = input.int(21, "Medium EMA", minval=1, maxval=200)
ma3_length = input.int(50, "Slow EMA", minval=1, maxval=200)
ma4_length = input.int(200, "Very Slow SMA", minval=1, maxval=200)

// Calculate Moving Averages
ema_fast = ta.ema(close, ma1_length)
ema_medium = ta.ema(close, ma2_length)
ema_slow = ta.ema(close, ma3_length)
sma_very_slow = ta.sma(close, ma4_length)

// Plot with different colors
plot(ema_fast, "Fast EMA 9", color=color.new(color.cyan, 0), linewidth=2)
plot(ema_medium, "Medium EMA 21", color=color.new(color.blue, 0), linewidth=2)
plot(ema_slow, "Slow EMA 50", color=color.new(color.orange, 0), linewidth=2)
plot(sma_very_slow, "Very Slow SMA 200", color=color.new(color.red, 0), linewidth=1)

// Background color when price is above/below MAs
bgcolor(close > ema_fast and close > ema_medium ? color.new(color.green, 95) : na, title="Bullish Zone")
bgcolor(close < ema_fast and close < ema_medium ? color.new(color.red, 95) : na, title="Bearish Zone")
